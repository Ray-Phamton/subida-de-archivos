#include <gtk/gtk.h>

static void on_button_clicked(GtkWidget *button, gpointer user_data) {
    const gchar *button_name = (const gchar *)user_data;
    g_print("El bot√≥n %s fue presionado\n", button_name);

    if (g_strcmp0(button_name, "Cerrar") == 0) {
        gtk_main_quit();
    }
}

int main(int argc, char *argv[]) {
    gtk_init(&argc, &argv);

    GtkWidget *window = gtk_window_new(GTK_WINDOW_TOPLEVEL);
    gtk_window_set_title(GTK_WINDOW(window), "Mi primera ventana GTK");

    GtkWidget *button_hola = gtk_button_new_with_label("Hola");
    g_signal_connect(button_hola, "clicked", G_CALLBACK(on_button_clicked), "Hola");
    gtk_container_add(GTK_CONTAINER(window), button_hola);

    GtkWidget *button_cerrar = gtk_button_new_with_label("Cerrar");
    g_signal_connect(button_cerrar, "clicked", G_CALLBACK(on_button_clicked), "Cerrar");
    gtk_container_add(GTK_CONTAINER(window), button_cerrar);

    gtk_widget_show_all(window);

    gtk_main();

    return 0;
}
